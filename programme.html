      #include <Servo.h>

      Servo monservo;

      byte angle=0;
      const byte bouton1=12 ;
      boolean etatbouton1=0;
      const int pot1 = A0;
      const int pot2 = A1;
      int vitesse;
      int volant;

      void setup()
      {
       monservo.attach(3);
       Serial.begin(9600);
       pinMode(bouton1, INPUT);
      }

      void loop()
      {
       vitesse=analogRead(pot1);
       volant=analogRead(pot2);
       etatbouton1=digitalRead(bouton1);
       Serial.println(vitesse);
       Serial.println(volant);
       if(etatbouton1==HIGH)
       {
         monservo.write(90);
         delay(10);
         etatbouton1==LOW;
       }
       if(vitesse>900)
       {
         monservo.write(90);
         delay(10);
       }
       if(volant>700||volant<300)
       {
         monservo.write(90);
         delay(10);
       }
       if(volant<700&&volant>300&&vitesse<900&&etatbouton1==LOW)
       {
         monservo.write(0);
         delay(10);
       }
      }
